43  /**
44   * This class contains utilities to delegate network manipulation. It hides the
45   * management for the current upload policy connection parameters.<BR>
46   * This class goes through the following states, stored in the private
47   * connectionStatus attribute: <DIR> <LI>STATUS_NOT_INITIALIZED: default status
48   * when the instance in created. Only available action:
49   * {@link #initRequest(URL, String, boolean, boolean)} <LI>
50   * STATUS_BEFORE_SERVER_CONNECTION: the instance is initialized, and the caller
51   * may begin writing the request to this HttpConnectionHelper. All data written
52   * to it, will be stored in a {@link ByteArrayEncoderHTTP}. The connection
53   * switches to this status when the
54   * {@link #initRequest(URL, String, boolean, boolean)} is called. <LI>
55   * STATUS_WRITING_REQUEST: The network connection to the server is now opened.
56   * The content of the ByteArrayEncoderHTTP has been sent to the server. All
57   * subsequent calls to write methods will directly write on the socket to the
58   * server. The {@link #sendRequest()} method changes the connection to this
59   * status. <LI>STATUS_READING_RESPONSE: The request to the server has been
60   * totally written. No more calls to the write methods are allowed. The
61   * {@link #readHttpResponse()} is responsible to put the HttpConnectionHelper to
62   * this status. <LI>STATUS_CONNECTION_CLOSED: The response has been read. All
63   * getters can be called, to get information about the server response. The only
64   * other method allowed is the
65   * {@link #initRequest(URL, String, boolean, boolean)}, to start a new request
66   * to the server. Using the same HttpConnectionHelper allows to use the same
67   * network connection, when the allowHttpPersistent applet parameter is used.
68   * </DIR>
69   * 
70   * @author etienne_sf
71   */
72  public class HTTPConnectionHelper extends OutputStream {

94      /**
95       * Indicates that the network connection to the server is opened, but the
96       * request has not been totally sent to the server. All data sent to the
97       * HttpConnectionHelper with write methods is sent to the network
98       * connection, that is: to the current OutputStream. <BR>
99       * That is: the ByteArrayEncoder is now read only (closed).
100      */
101     final static private int STATUS_WRITING_REQUEST = 2;
102 
103     /**
104      * Indicates that the network connection to the server is opened, but the
105      * request has not been totally sent to the server. All data sent to the
106      * HttpConnectionHelper with write methods is sent to the network
107      * connection, that is: to the current OutputStream. <BR>
108      * That is: the ByteArrayEncoder is now read only (closed).
109      */
110     final static private int STATUS_READING_RESPONSE = 3;
111 
112     /**
113      * Indicates that the network connection to the server is now closed, that
114      * is: we've written the request, read the response, and free the server
115      * connection. If the keepAlive parameter is used, the connection may remain
116      * opened for the next request. <BR>
117      * No more action may be done on this connection helper, out of reading
118      * data, until the application do a call to
119      * {@link #initRequest(URL, String, boolean, boolean)}.
120      */
121     final static private int STATUS_CONNECTION_CLOSED = 4;

243     /**
244      * The standard constructor for this class.
245      * 
246      * @param url The target URL
247      * @param bChunkEnabled Indicates if chunkUpload is enabled for this query.
248      *            Put false, if non chunked request or if it is not relevant.
249      * @param method The HTTP method (POST, GET, HEAD...)
250      * @param bLastChunk Indicates whether this chunk is the last one. Put true,
251      *            if non chunked request or if it is not relevant.
252      * @throws JUploadIOException
253      */
254     public synchronized void initRequest(URL url, String method,
255             boolean bChunkEnabled, boolean bLastChunk)
256             throws JUploadIOException {
257         // This method expects that the connection has not been initialized yet,
258         // or that the previous request is finished.
259         if (this.connectionStatus != STATUS_NOT_INITIALIZED
260                 && this.connectionStatus != STATUS_CONNECTION_CLOSED) {
261             throw new JUploadIOException(
262                     "Bad status of the HttpConnectionHelper in initRequest: "
263                             + getStatusLabel());
264         }
265 
266         // Clean any current request.
267         if (isKeepAlive()) {
268             dispose();
269         }
270 
271         // Load the new parameters.
272         this.url = url;
273         this.method = method;
274         this.bChunkEnabled = bChunkEnabled;
275         this.bLastChunk = bLastChunk;
276         // We will write to the local ByteArrayEncoder, until a connection to
277         // the server is opened.
278         initByteArrayEncoder();
279 
280         // Ok, the HttpConnectionHelper is now ready to get write commands.
281         this.connectionStatus = STATUS_BEFORE_SERVER_CONNECTION;
282     }

304     /**
305      * Closes the byteArrayEncoder, create the socket (or not, depending on the
306      * current uploadPolicy, and upload history), send the request, and create
307      * the InputStream to read the server response.
308      * 
309      * @throws JUploadIOException
310      */
311     public synchronized void sendRequest() throws JUploadIOException {
312         // This method expects that the connection is writing data to the
313         // server.
314         if (this.connectionStatus != STATUS_BEFORE_SERVER_CONNECTION) {
315             throw new JUploadIOException(
316                     "Bad status of the HttpConnectionHelper in sendRequest: "
317                             + getStatusLabel());
318         }
319 
320         try {
321             // We've finished with the current encoder.
322             if (!this.byteArrayEncoder.isClosed()) {
323                 this.byteArrayEncoder.close();
324             }
325 
326             // Let's clear any field that could have been read in a previous
327             // step:
328             this.httpInputStreamReader = null;
329 
330             // Only connect, if sock is null!!
331             // ... or if we don't persist HTTP connections (patch for IIS, based
332             // on Marc Reidy's patch)
333             if (this.socket == null
334                     || !this.uploadPolicy.getAllowHttpPersistent()) {
335                 this.socket = new HttpConnect(this.uploadPolicy).connect(
336                         this.url, this.proxy);
337                 this.outputStream = new DataOutputStream(
338                         new BufferedOutputStream(this.socket.getOutputStream()));
339                 this.inputStream = new PushbackInputStream(
340                         this.socket.getInputStream(), 1);
341             }
342 
343             // Send http request to server
344             this.outputStream
345                     .write(this.byteArrayEncoder.getEncodedByteArray());
346 
347             // The request has been sent. The current ByteArrayEncoder is now
348             // useless. A new one is to be created for the next request.
349             this.connectionStatus = STATUS_WRITING_REQUEST;
350 
351         } catch (IOException e) {
352             throw new JUploadIOException("Unable to open socket", e);
353         } catch (KeyManagementException e) {
354             throw new JUploadIOException("Unable to open socket", e);
355         } catch (UnrecoverableKeyException e) {
356             throw new JUploadIOException("Unable to open socket", e);
357         } catch (NoSuchAlgorithmException e) {
358             throw new JUploadIOException("Unable to open socket", e);
359         } catch (KeyStoreException e) {
360             throw new JUploadIOException("Unable to open socket", e);
361         } catch (CertificateException e) {
362             throw new JUploadIOException("Unable to open socket", e);
363         } catch (IllegalArgumentException e) {
364             throw new JUploadIOException("Unable to open socket", e);
365         }
366 
367     }

369     /**
370      * Releases all reserved resources.
371      * 
372      * @throws JUploadIOException
373      */
374     public synchronized void dispose() throws JUploadIOException {
375         // A try to properly clean the connection to the server.
376         try {
377             // Let's shutdown the output, to free the server connection. Only
378             // valid for non https connections.
379             if (this.socket != null && !this.useSSL
380                     && !this.socket.isOutputShutdown()) {
381                 this.socket.shutdownOutput();
382             }
383         } catch (IOException e) {
384             throw new JUploadIOException(e);
385         }
386         try {
387             if (this.socket != null && !this.useSSL
388                     && !this.socket.isInputShutdown()) {
389                 this.socket.shutdownInput();
390             }
391         } catch (IOException e) {
392             throw new JUploadIOException(e);
393         }
394 
395         try {
396             if (this.outputStream != null) {
397                 this.outputStream.close();
398             }
399         } catch (IOException e) {
400             throw new JUploadIOException(e);
401         } finally {
402             this.outputStream = null;
403         }
404 
405         try {
406             if (this.inputStream != null) {
407                 this.inputStream.close();
408             }
409         } catch (IOException e) {
410             throw new JUploadIOException(e);
411         } finally {
412             this.inputStream = null;
413         }
414 
415         try {
416             if (this.socket != null) {
417                 if (!this.socket.isClosed()) {
418                     this.socket.close();
419                 }
420             }
421         } catch (IOException e) {
422             throw new JUploadIOException(e);
423         } finally {
424             this.socket = null;
425         }
426     }

522     /**
523      * Append bytes to the current query. The bytes will be written to the
524      * current ByteArrayEncoder if the the connection to the server is not open,
525      * or directly to the server if the connection is opened.
526      * 
527      * @param b The byte to send to the server.
528      * @return Returns the current HttpConnectionHelper, to allow coding like
529      *         StringBuffers: a.append(b).append(c);
530      * @throws JUploadIOException
531      */
532     public synchronized HTTPConnectionHelper append(int b)
533             throws JUploadIOException {
534         if (this.connectionStatus == STATUS_BEFORE_SERVER_CONNECTION) {
535             this.byteArrayEncoder.append(b);
536         } else if (this.connectionStatus == STATUS_WRITING_REQUEST) {
537             try {
538                 this.outputStream.write(b);
539             } catch (IOException e) {
540                 throw new JUploadIOException(e.getClass().getName()
541                         + " while writing to httpDataOut", e);
542             }
543         } else {
544             throw new JUploadIOException(
545                     "Wrong status in HTTPConnectionHelper.write() ["
546                             + getStatusLabel() + "]");
547         }
548         return this;
549     }

580 
581         if (this.connectionStatus == STATUS_BEFORE_SERVER_CONNECTION) {
582             this.byteArrayEncoder.append(bytes);
583         } else if (this.connectionStatus == STATUS_WRITING_REQUEST) {
584             try {
585                 this.outputStream.write(bytes, off, len);
586             } catch (IOException e) {
587                 throw new JUploadIOException(e.getClass().getName()
588                         + " while writing to httpDataOut", e);

590         } else {
591             throw new JUploadIOException(

593                             + getStatusLabel() + "]");
594         }
595 
596         if (this.uploadPolicy.getDebugLevel() > 100) {
597             this.uploadPolicy
598                     .displayDebug(
599                             "[HTTPConnectionHelper append(byte[],int,int)] ("
600                                     + len
601                                     + " bytes appended to "
602                                     + (this.connectionStatus == STATUS_BEFORE_SERVER_CONNECTION ? " current ByteArrayEncoder"
603                                             : " socket") + ")", 101);
604         }
605 
606         return this;
607     }

609     /**
610      * write a string to the current HTTP request.
611      * 
612      * @param str The string to write
613      * @return The current HTTPConnectionHelper
614      * @throws JUploadIOException If any problem occurs during the writing
615      *             operation.
616      * @see #append(byte[])
617      */
618     public synchronized HTTPConnectionHelper append(String str)
619             throws JUploadIOException {
620         this.uploadPolicy.displayDebug("[HTTPConnectionHelper append] " + str,
621                 70);
622         if (this.connectionStatus == STATUS_BEFORE_SERVER_CONNECTION) {
623             this.byteArrayEncoder.append(str);
624         } else if (this.connectionStatus == STATUS_WRITING_REQUEST) {
625             ByteArrayEncoder bae = new ByteArrayEncoderHTTP(this.uploadPolicy,
626                     this.byteArrayEncoder.getBoundary(),
627                     this.byteArrayEncoder.getEncoding());
628             bae.append(str);
629             bae.close();
630             this.append(bae);
631         }
632         return this;
633     }

653     /**
654      * Read the response of the server. This method delegates the work to the
655      * HTTPInputStreamReader. handles the chunk HTTP response.
656      * 
657      * @return The HTTP status. Should be 200, when everything is right.
658      * @throws JUploadException
659      */
660     public synchronized int readHttpResponse() throws JUploadException {
661         // This method expects that the connection is writing data to the
662         // server.
663         if (this.connectionStatus != STATUS_WRITING_REQUEST) {
664             throw new JUploadIOException(
665                     "Bad status of the HttpConnectionHelper in readHttpResponse: "
666                             + getStatusLabel());
667         }
668         this.connectionStatus = STATUS_READING_RESPONSE;
669 
670         // Let's connect in InputStream to read this server response.
671         if (this.httpInputStreamReader == null) {
672             this.httpInputStreamReader = new HTTPInputStreamReader(this,
673                     this.uploadPolicy);
674         }
675 
676         // Let's do the job
677         try {
678             this.outputStream.flush();
679         } catch (IOException ioe) {
680             throw new JUploadIOException("flushing outputStream, in "
681                     + getClass().getName() + ".readHttpResponse()");
682         }
683         this.httpInputStreamReader.readHttpResponse();
684 
685         if (this.httpInputStreamReader.gotClose) {
686             // RFC 2868, section 8.1.2.1
687             dispose();
688         }
689 
690         // We got the response
691         this.connectionStatus = STATUS_CONNECTION_CLOSED;
692 
693         // FIXME Should close the connection, from time to time...
694 
695         //
696         return this.httpInputStreamReader.gethttpStatusCode();
697     }

703     /**
704      * creating of a new {@link ByteArrayEncoderHTTP}, and initializing of the
705      * following header items: First line (POST currentProtocol URI), Host,
706      * Connection, Keep-Alive, Proxy-Connection.
707      * 
708      * @throws JUploadIOException
709      */
710     private synchronized void initByteArrayEncoder() throws JUploadIOException {
711         if (this.byteArrayEncoder != null && !this.byteArrayEncoder.isClosed()) {
712             this.byteArrayEncoder.close();
713             this.byteArrayEncoder = null;
714         }
715         this.byteArrayEncoder = new ByteArrayEncoderHTTP(this.uploadPolicy,
716                 this.boundary);
717         this.connectionStatus = STATUS_BEFORE_SERVER_CONNECTION;
718         this.proxy = null;
719         try {
720             this.proxy = ProxySelector.getDefault().select(this.url.toURI())
721                     .get(0);
722         } catch (URISyntaxException e) {
723             throw new JUploadIOException("Error while managing url "
724                     + this.url.toExternalForm(), e);
725         }
726         this.useProxy = ((this.proxy != null) && (this.proxy.type() != Proxy.Type.DIRECT));
727         this.useSSL = this.url.getProtocol().equals("https");
728 
729         // Header: Request line
730         // Let's clear it. Useful only for chunked uploads.
731         this.byteArrayEncoder.append(this.method);
732         this.byteArrayEncoder.append(" ");
733         this.uploadPolicy.displayDebug("[initByteArrayEncoder] proxy=" + proxy
734                 + ", proxy.type=" + this.proxy.type() + ", useProxy="
735                 + useProxy + ", url.host=" + this.url.getHost() + ", url.port="
736                 + this.url.getPort(), 80);
737         if (this.useProxy && (!this.useSSL)) {
738             // with a proxy we need the absolute URL, but only if not
739             // using SSL. (with SSL, we first use the proxy CONNECT method,
740             // and then a plain request.)
741             this.byteArrayEncoder.append(this.url.getProtocol()).append("://")
742                     .append(this.url.getHost());
743             // TODO port in proxy mode: to be tested !
744             if (this.url.getPort() > 0) {
745                 this.byteArrayEncoder.append(":").append(
746                         Integer.toString(this.url.getPort()));
747             }
748         }
749         this.byteArrayEncoder.append(this.url.getPath());
750 
751         // Append the query params.
752         // TODO: This probably can be removed as we now have everything in POST
753         // data. However in order to be
754         // backwards-compatible, it stays here for now. So we now provide
755         // *both* GET and POST params.
756         if (null != this.url.getQuery() && !"".equals(this.url.getQuery()))
757             this.byteArrayEncoder.append("?").append(this.url.getQuery());
758 
759         this.byteArrayEncoder.append(" ")
760                 .append(this.uploadPolicy.getServerProtocol()).append("\r\n");
761 
762         // Header: General
763         this.byteArrayEncoder.append("Host: ").append(this.url.getHost());
764         if (this.url.getPort() >0) {
765             this.byteArrayEncoder.append(":").append(Integer.toString(this.url.getPort()));
766         }
767         this.byteArrayEncoder.append("\r\nAccept: */*\r\n");
768         
769         // We do not want gzipped or compressed responses, so we must
770         // specify that here (RFC 2616, Section 14.3)
771         this.byteArrayEncoder.append("Accept-Encoding: identity\r\n");
772 
773         // Seems like the Keep-alive doesn't work properly, at least on my
774         // local dev (Etienne).
775         if (!this.uploadPolicy.getAllowHttpPersistent()) {
776             this.byteArrayEncoder.append("Connection: close\r\n");
777         } else {
778             if (!this.bChunkEnabled
779                     || this.bLastChunk
780                     || this.useProxy
781                     || !this.uploadPolicy.getServerProtocol()
782                             .equals("HTTP/1.1")) { // RFC 2086, section 19.7.1
783                 this.byteArrayEncoder.append("Connection: close\r\n");
784             } else {
785                 this.byteArrayEncoder.append("Keep-Alive: 300\r\n");
786                 if (this.useProxy)
787                     this.byteArrayEncoder
788                             .append("Proxy-Connection: keep-alive\r\n");
789                 else
790                     this.byteArrayEncoder.append("Connection: keep-alive\r\n");
791             }
792         }
793 
794         // Get specific headers for this upload.
795         this.uploadPolicy.onAppendHeader(this.byteArrayEncoder);
796     }

798     /**
799      * Indicates whether the current socket should be reused ... if any.
800      */
801     private boolean isKeepAlive() {
802         if (this.socket == null) {
803             return false;
804         } else if (!this.uploadPolicy.getAllowHttpPersistent()) {
805             return false;
806         } else {
807             if (!this.bChunkEnabled
808                     || this.bLastChunk
809                     || this.useProxy
810                     || !this.uploadPolicy.getServerProtocol()
811                             .equals("HTTP/1.1")) { // RFC 2086, section 19.7.1
812                 return false;
813             } else {
814                 return true;
815             }
816         }

823      * @return The calculated boundary.

825     private final String calculateRandomBoundary() {
826         StringBuffer sbRan = new StringBuffer(11);
827         sbRan.append("-----------------------------");
828         String alphaNum = "1234567890abcdefghijklmnopqrstuvwxyz";
829         int num;
830         for (int i = 0; i < 11; i++) {
831             num = (int) (Math.random() * (alphaNum.length() - 1));
