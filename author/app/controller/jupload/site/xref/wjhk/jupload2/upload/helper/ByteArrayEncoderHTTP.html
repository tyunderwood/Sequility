33  /**
34   * This class is a utility, which provide easy encoding for HTTP queries. The
35   * way to use this class is:
36   * <OL TYPE=1>
37   * <LI>Instantiate a new object
38   * <LI>Append data to it, using the append methods. Available for: String,
39   * byte[], other ByteArrayEncode...
40   * <LI>Close the stream. This will prevent any new data to be appended to it.
41   * The encoded length can now be calculated.
42   * <LI>Get the encoded length.
43   * <LI>Get the encoded byte array
44   * </OL>
45   * 
46   * @author etienne_sf
47   * 
48   */

152     /**
153      * @throws JUploadIOException
154      * @see wjhk.jupload2.upload.helper.ByteArrayEncoder#close()
155      */
156     synchronized public void close() throws JUploadIOException {
157         if (isClosed()) {
158             throw new JUploadIOException(
159                     "Trying to close an already closed ByteArrayEncoded");
160         }
161         try {
162             this.writer.close();
163         } catch (IOException e) {
164             throw new JUploadIOException(e);
165         }
166         this.encodedByteArray = this.baos.toByteArray();
167         this.encodedLength = this.encodedByteArray.length;
168         this.closed = true;
169     }

217     /** {@inheritDoc} */
218     public ByteArrayEncoder appendTextProperty(String name, String value,
219             int index) throws JUploadIOException {
220         String propertySuffix = "";
221         // if an index is given, we may have to suffix the name of the upload
222         // parameter, depending on the value of httpUploadParameterType
223         if (index >= 0) {
224             if (this.uploadPolicy.getHttpUploadParameterType().equals(
225                     UploadPolicy.HTTPUPLOADPARAMETERTYPE_ARRAY)) {
226                 propertySuffix = "[]";
227             } else if (this.uploadPolicy.getHttpUploadParameterType().equals(
228                     UploadPolicy.HTTPUPLOADPARAMETERTYPE_ITERATION)) {
229                 propertySuffix = Integer.toString(index);
230             }
231         }
232         this.append(this.bound).append("\r\n");
233         this.append("Content-Disposition: form-data; name=\"").append(name)
234                 .append(propertySuffix).append("\"\r\n");
235         this.append("Content-Transfer-Encoding: 8bit\r\n");
236         this.append("Content-Type: text/plain; charset=").append(this.getEncoding())
237                 .append("\r\n");
238         // An empty line before the actual value.
239         this.append("\r\n");
240         // And then, the value!
241         this.append(value).append("\r\n");
242 
243         return this;
244     }
245 
246     /** {@inheritDoc} */

252     /** {@inheritDoc} */
253     public ByteArrayEncoder appendFormVariables(String formname)
254             throws JUploadIOException {
255         String action = "Entering ByteArrayEncoderHTTP.appendFormVariables() [html form: "
256                 + formname + "]";
257         try {
258             this.uploadPolicy.displayDebug(action, 80);
259             // TODO Do not use getApplet() anymore, here.
260             action = "win = netscape.javascript.JSObject.getWindow";
261             this.uploadPolicy.displayDebug("Before " + action + 
262                     " (this.uploadPolicy.getContext().getApplet(): "
263                     + this.uploadPolicy.getContext().getApplet() + ")"
264                     , 80);
265             netscape.javascript.JSObject win = netscape.javascript.JSObject
266                     .getWindow(this.uploadPolicy.getContext().getApplet());
267             this.uploadPolicy.displayDebug("After " + action + 
268                     " (win: " + win + ")", 80);
269 
270             action = "o = win.eval";
271             Object o = win.eval("document.forms[\"" + formname
272                     + "\"].elements.length");
273             if (o instanceof Number) {
274                 int len = ((Number) o).intValue();
275                 if (len <= 0) {
276                     this.uploadPolicy.displayWarn("The specified form \""
277                             + formname + "\" does not contain any elements.");
278                 }
279                 int i;
280                 for (i = 0; i < len; i++) {
281                     try {
282                         action = "name = win.eval";
283                         Object name = win.eval("document.forms[\"" + formname
284                                 + "\"][" + i + "].name");
285                         action = "value = win.eval";
286                         Object value = win.eval("document.forms[\"" + formname
287                                 + "\"][" + i + "].value");
288                         action = "elementType = win.eval";
289                         Object elementType = win.eval("document.forms[\""
290                                 + formname + "\"][" + i + "].type");
291                         action = "elementClass = win.eval";
292                         // elementClass seems to be not supported by IE7
293                         // The next line prevents formData to be sent to the
294                         // server, when formData is used on IE7. Works fine with
295                         // firefox.
296                         // Object elementClass = win.eval("document." + formname
297                         // + "[" + i + "].class");
298                         action = "etype = win.eval";
299                         Object etype = win.eval("document.forms[\"" + formname
300                                 + "\"][" + i + "].type");
301                         if (etype instanceof String) {
302                             String t = (String) etype;
303                             if (t.equals("checkbox") || t.equals("radio")) {
304                                 action = "on = win.eval";
305                                 Object on = win.eval("document.forms[\""
306                                         + formname + "\"][" + i + "].checked");
307                                 if (on instanceof Boolean) {
308                                     // Skip unchecked checkboxes and
309                                     // radiobuttons
310                                     if (!((Boolean) on).booleanValue()) {
311                                         this.uploadPolicy
312                                                 .displayDebug(
313                                                         "  [ByteArrayEncoder.appendFormVariables] Skipping unchecked checkboxes and radiobuttons",
314                                                         80);
315                                         // TODO Do not use getApplet() anymore,
316                                         // here.
317                                         continue;
318                                     }
319                                 }
320                             }
321                         }
322                         if (name instanceof String
323                                 && ((String) name).length() > 0) {
324                             if (value instanceof String) {
325                                 this.uploadPolicy.displayDebug(
326                                         "  [ByteArrayEncoder.appendFormVariables] Adding formdata element num "
327                                                 + i + " (name: " + name
328                                                 + ", value: " + value
329                                                 + ", type: " + elementType
330                                                 /* + ", class: " + elementClass */
331                                                 + ")", 80);
332                                 this.appendTextProperty((String) name,
333                                         (String) value, -1);
334                             } else {
335                                 this.uploadPolicy
336                                         .displayWarn("  [ByteArrayEncoder.appendFormVariables] Value must be an instance of String (name: "
337                                                 + name
338                                                 + ", value: )"
339                                                 + value
340                                                 + ")");
341                             }
342                         } else {
343                             this.uploadPolicy
344                                     .displayWarn("  [ByteArrayEncoder.appendFormVariables] Name must be an instance of String (name: "
345                                             + name + ", value: )" + value + ")");
346                         }
347                     } catch (netscape.javascript.JSException e1) {
348                         this.uploadPolicy
349                                 .displayWarn(e1.getStackTrace()[1]
350                                         + ": got JSException in ByteArrayEncoderHTTP.appendFormVariables() [html form: "
351                                         + formname
352                                         + "] - bailing out (action: " + action
353                                         + ")");
354                         i = len;
355                     }
356                 }
357             } else {
358                 this.uploadPolicy
359                         .displayWarn("  [ByteArrayEncoder.appendFormVariables] The specified form \""
360                                 + formname
361                                 + "\" could not be found (or has no element).");
362             }
363         } catch (netscape.javascript.JSException e) {
364             this.uploadPolicy.displayDebug(e.getStackTrace()[1]
365                     + ": No JavaScript available (action: " + action + ")", 10);
366         }
367 
368         this.uploadPolicy
369                 .displayDebug(
370                         "  [ByteArrayEncoder.appendFormVariables] End of ByteArrayEncoderHTTP.appendFormVariables()",
371                         80);
372         return this;
373     }

415     /** {@inheritDoc} */
416     public String getString() throws JUploadIOException {
417         if (!isClosed()) {
418             throw new JUploadIOException(

421         try {
