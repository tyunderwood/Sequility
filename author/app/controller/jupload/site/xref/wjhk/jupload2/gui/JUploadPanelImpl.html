191 		if (this.browseButton == null) {

290 	/** {@inheritDoc} */
291 	public void doBrowse() {
292 		// If the file chooser was not created, we create it now.
293 		if (null == this.fileChooser) {
294 			// Setup File Chooser.
295 			try {
296 				this.uploadPolicy.displayDebug(
297 						"Before this.uploadPolicy.createFileChooser()", 80);
298 				this.fileChooser = this.uploadPolicy.createFileChooser();
299 				this.uploadPolicy.displayDebug(
300 						"After this.uploadPolicy.createFileChooser()", 80);
301 			} catch (Exception e) {
302 				this.uploadPolicy.displayErr(e);
303 			}
304 
305 		}
306 
307 		// If the fileChooser succeed (should be the case), we go on.
308 		if (null != this.fileChooser) {
309 			try {
310 				int ret = this.fileChooser.showOpenDialog(new Frame());
311 				if (JFileChooser.APPROVE_OPTION == ret)
312 					this.filePanel.addFiles(
313 							this.fileChooser.getSelectedFiles(),
314 							this.fileChooser.getCurrentDirectory());
315 				// We stop any running task for the JUploadFileView
316 				this.uploadPolicy.setCurrentBrowsingDirectory(this.fileChooser
317 						.getCurrentDirectory().getAbsolutePath());
318 				this.fileChooser.shutdownNow();
319 			} catch (Exception ex) {
320 				this.uploadPolicy.displayErr(ex);
321 			}
322 		}
323 	}

343 	/** {@inheritDoc} */
344 	public void doStartUpload() {
345 		// Check that the upload is ready (we ask the uploadPolicy. Then,
346 		// we'll call beforeUpload for each
347 		// FileData instance, that exists in allFiles[].
348 
349 		// ///////////////////////////////////////////////////////////////////////////////////////////////
350 		// IMPORTANT: It's up to the UploadPolicy to explain to the user
351 		// that the upload is not ready!
352 		// ///////////////////////////////////////////////////////////////////////////////////////////////
353 		try {
354 			if (this.uploadPolicy.beforeUpload()) {
355 			    // When called from the javascript handler, there may be no
356 			    // file in the filelist. 
357 			    if (getFilePanel().getFilesLength() > 0) {
358     				// The FileUploadManagerThread will manage everything around
359     				// upload, including GUI part.
360     				this.fileUploadManagerThread = new FileUploadManagerThreadImpl(
361     						this.uploadPolicy);
362     				this.fileUploadManagerThread.start();
363 			    }
364 			} // if isIploadReady()
365 		} catch (Exception e) {
366 			// If an exception occurs here, it fails silently. The exception is
367 			// thrown to the AWT event dispatcher.
368 			this.uploadPolicy.displayErr(e.getClass().getName()
369 					+ " in JUploadPanelImpl.doStartUpload()", e);
370 		}
371 	}

382 	/**
383 	 * @see java.awt.event.ActionListener#actionPerformed(java.awt.event.ActionEvent)
384 	 */
385 	public void actionPerformed(ActionEvent e) {
386 		// Let's log some info.
387 		this.uploadPolicy.displayDebug("Action : " + e.getActionCommand(), 1);
388 
389 		final String actionPaste = (String) TransferHandler.getPasteAction()
390 				.getValue(Action.NAME);
391 
392 		if (e.getActionCommand().equals(actionPaste)) {
393 			Action a = getActionMap().get(actionPaste);
394 			if (a != null) {
395 				a.actionPerformed(new ActionEvent(this.filePanel,
396                     ActionEvent.ACTION_PERFORMED, e.getActionCommand()));
397                 this.uploadPolicy.afterFileDropped(new DropTargetDropEvent(new DropTarget(
398                     this.filePanel.getDropComponent(), this.dndListener).getDropTargetContext(),
399                     new Point(), DnDConstants.ACTION_MOVE, DnDConstants.ACTION_COPY_OR_MOVE ));
400 			}
401 		} else if (e.getActionCommand() == this.browseButton.getActionCommand()) {
402 			doBrowse();
403 		} else if (e.getActionCommand() == this.removeButton.getActionCommand()) {
404 			// Remove clicked
405 			doRemove();
406 		} else if (e.getActionCommand() == this.removeAllButton
407 				.getActionCommand()) {
408 			// Remove All clicked
409 			doRemoveAll();
410 		} else if (e.getActionCommand() == this.uploadButton.getActionCommand()) {
411 			// Upload clicked
412 			doStartUpload();
413 		} else if (e.getActionCommand() == this.stopButton.getActionCommand()) {
414 			// We request the thread to stop its job.
415 			doStopUpload();
416 		}
417 		// focus the table. This is necessary in order to enable mouse
418 		// events
419 		// for triggering tooltips.
420 		this.filePanel.focusTable();
421 	}

469 	/** {@inheritDoc} */
470 	public boolean maybeOpenPopupMenu(MouseEvent mouseEvent) {
471 		// Should we open one out of the numerous (2!) popup menus ?
472 		if (mouseEvent.isPopupTrigger()) {
473 			if ((mouseEvent.getModifiersEx() & InputEvent.CTRL_DOWN_MASK) == InputEvent.CTRL_DOWN_MASK) {
474 				// We open the debug menu
475 				if (this.jUploadDebugPopupMenu != null) {
476 					this.jUploadDebugPopupMenu.show(mouseEvent.getComponent(),
477 							mouseEvent.getX(), mouseEvent.getY());
478 					return true;
479 				}
480 			} else {
481 				// Let's open the main popup menu
482 				if (this.jUploadMainPopupMenu != null) {
483 					this.jUploadMainPopupMenu.show(mouseEvent.getComponent(),
484 							mouseEvent.getX(), mouseEvent.getY());
485 					return true;
486 				}
487 			}
488 		}
489 		return false;
490 	}

492 	/** {@inheritDoc} */
493 	public void updateButtonState() {
494 		if (this.fileUploadManagerThread != null
495 				&& this.fileUploadManagerThread.isAlive()
496 				&& !this.fileUploadManagerThread.isUploadFinished()) {
497 			// An upload is running on.
498 			this.browseButton.setEnabled(false);
499 			this.removeButton.setEnabled(false);
500 			this.removeAllButton.setEnabled(false);
501 			this.uploadButton.setEnabled(false);
502 			this.stopButton.setEnabled(true);
503 			// Let's delegate any additional work to the upload policy.
504 			this.uploadPolicy
505 					.updateButtonState(UploadPolicy.EXEC_STATUS_UPLOADING);
506 		} else {
507 			// No upload running on.
508 			this.browseButton.setEnabled(true);
509 			this.stopButton.setEnabled(false);
510 
511 			boolean enabled = (this.filePanel.getFilesLength() > 0);
512 			this.removeButton.setEnabled(enabled);
513 			this.removeAllButton.setEnabled(enabled);
514 			this.uploadButton.setEnabled(enabled);
515 			// Let's delegate any additional work to the upload policy.
516 			this.uploadPolicy.updateButtonState(UploadPolicy.EXEC_STATUS_READY);
517 		}
